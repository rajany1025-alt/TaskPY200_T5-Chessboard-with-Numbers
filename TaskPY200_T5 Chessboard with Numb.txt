# TaskPY200_T5: Chessboard with Numbers

# Take input
N = int(input("Enter N: "))

# Loop through rows
for row in range(1, N + 1):

    # Loop through columns
    for col in range(1, N + 1):

        # Check diagonal condition first
        if row == col:
            print("X", end=" ")

        # Check even condition
        elif (row + col) % 2 == 0:
            print("1", end=" ")

        # Else print 0
        else:
            print("0", end=" ")

    # Move to next line after each row
    print()
